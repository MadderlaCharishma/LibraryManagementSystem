    id INT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    author VARCHAR(255) NOT NULL,
    category VARCHAR(100) NOT NULL,
    isIssued BOOLEAN DEFAULT FALSE,
    issueDate DATE DEFAULT NULL
);
